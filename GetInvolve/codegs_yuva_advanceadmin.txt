// ===== MAIN HANDLER =====
/**
 * Handle incoming requests from the frontend
 */
function doPost(e) {
    try {
        const payload = JSON.parse(e.postData.contents || '{}');

        // Notification/email methods do not need counter visitor/session fields.
        if (payload.action === 'notify') {
            return handleNotify(payload);
        }

        // Validate counter request
        if (!validateRequest(payload)) {
            return createResponse(false, 'Invalid request', 400);
        }

        let result;

        switch (payload.action) {
            case 'recordRefresh':
                result = recordPageRefresh(payload);
                break;
            case 'recordUniqueVisit':
                result = recordUniqueVisitor(payload);
                break;
            case 'getCounters':
                result = getCounters(payload);
                break;
            case 'reset':
                result = resetCounters(payload);
                break;
            default:
                return createResponse(false, 'Unknown action: ' + payload.action, 400);
        }

        return createResponse(result.success, result.message, result.statusCode || 200, result.data);
    } catch (error) {
        logError('doPost error: ' + error.toString());
        return createResponse(false, 'Server error: ' + error.toString(), 500);
    }
}

// ===== NOTIFY / EMAIL HANDLER =====
function handleNotify(payload) {
    try {
        const method = String(payload.method || '').trim();

        switch (method) {
            case 'sendVimarshCertificateEmails':
                return sendVimarshCertificateEmails(payload);
            default:
                return createResponse(false, 'Invalid notify method: ' + method, 400);
        }
    } catch (error) {
        logError('handleNotify error: ' + error.toString());
        return createResponse(false, 'Notify handler failed: ' + error.toString(), 500);
    }
}

function safeHtml_(value) {
    return String(value || '')

        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;')
        .replace(/'/g, '&#39;');
}

// ===== REQUEST VALIDATION =====
/**
 * Validate incoming request
 */
function validateRequest(payload) {
    // Check required fields
    if (!payload.action || !payload.visitorId || !payload.sessionToken) {
        return false;
    }

    // Check if visitor ID format is valid
    if (!payload.visitorId.toString().startsWith('visitor_')) {
        return false;
    }

    // Check if session token format is valid
    if (!payload.sessionToken.toString().startsWith('session_')) {
        return false;
    }

    return true;
}

// ===== COUNTER OPERATIONS =====
/**
 * Record page refresh
 */
function recordPageRefresh(payload) {
    try {
        // Get or create counter record
        let counterData = getCounterRecord();

        // Increment refresh count
        counterData.refresh_count = (counterData.refresh_count || 0) + 1;
        counterData.last_refresh_time = new Date().toISOString();

        // Update record
        const updateResult = updateCounterRecord(counterData);

        // Log the action
        logCounterAction(payload, 'refresh');

        return {
            success: updateResult.success,
            message: updateResult.success ? 'Refresh recorded' : 'Failed to update refresh count',
            statusCode: updateResult.success ? 200 : 500
        };
    } catch (error) {
        logError('recordPageRefresh error: ' + error.toString());
        return {
            success: false,
            message: 'Error recording refresh: ' + error.toString(),
            statusCode: 500
        };
    }
}

/**
 * Record unique visitor
 */
function recordUniqueVisitor(payload) {
    try {
        // Check if visitor already recorded today
        const existingVisit = checkExistingVisit(payload.visitorId);

        if (existingVisit) {
            return {
                success: false,
                message: 'Visitor already counted in this period',
                statusCode: 200
            };
        }

        // Get or create counter record
        let counterData = getCounterRecord();

        // Increment unique count
        counterData.unique_count = (counterData.unique_count || 0) + 1;
        counterData.last_unique_time = new Date().toISOString();

        // Update record
        const updateResult = updateCounterRecord(counterData);

        // Log the visitor
        logUniqueVisitor(payload.visitorId);

        // Log the action
        logCounterAction(payload, 'unique_visit');

        return {
            success: updateResult.success,
            message: updateResult.success ? 'Unique visitor recorded' : 'Failed to update unique count',
            statusCode: updateResult.success ? 200 : 500
        };
    } catch (error) {
        logError('recordUniqueVisitor error: ' + error.toString());
        return {
            success: false,
            message: 'Error recording unique visitor: ' + error.toString(),
            statusCode: 500
        };
    }
}

/**
 * Reset counters (admin only)
 */
function resetCounters(payload) {
    try {
        // Additional security check for reset
        if (payload.adminKey !== 'YOUR_ADMIN_KEY_HERE') {
            logError('Unauthorized reset attempt from: ' + payload.visitorId);

                success: false,
                message: 'Unauthorized',

        return {
            success: updateResult.success,
            message: updateResult.success ? 'Counters reset' : 'Failed to reset',
            statusCode: updateResult.success ? 200 : 500
        };
    } catch (error) {
        logError('resetCounters error: ' + error.toString());
        return {
            success: false,
            message: 'Error resetting counters: ' + error.toString(),
            statusCode: 500
        };
    }
}

// ===== DATABASE OPERATIONS =====
/**
 * Get the main counter record from Supabase
 */
function getCounterRecord() {
    const url = `${SUPABASE_URL}/rest/v1/${TABLE_NAME}?id=eq.1`;

    const options = {
        method: 'get',
        headers: {
            'apikey': SUPABASE_API_KEY,
            'Authorization': 'Bearer ' + SUPABASE_API_KEY,
            'Content-Type': 'application/json'
        },
        muteHttpExceptions: true
    };

    try {
        const response = UrlFetchApp.fetch(url, options);
        const data = JSON.parse(response.getContentText());

        if (Array.isArray(data) && data.length > 0) {
            return data[0];
        }

        // Create new record if doesn't exist
        return createCounterRecord();
    } catch (error) {
        logError('getCounterRecord error: ' + error.toString());
        throw error;
    }
}

    }
}

/**
 * Update counter record in Supabase
 */
function updateCounterRecord(data) {
    const url = `${SUPABASE_URL}/rest/v1/${TABLE_NAME}?id=eq.1`;

    data.updated_at = new Date().toISOString();

    const options = {
        method: 'patch',
        headers: {
            'apikey': SUPABASE_API_KEY,
            'Authorization': 'Bearer ' + SUPABASE_API_KEY,
            'Content-Type': 'application/json'
        },
        payload: JSON.stringify(data),
        muteHttpExceptions: true
    };

    try {
        const response = UrlFetchApp.fetch(url, options);
        return {
            success: response.getResponseCode() >= 200 && response.getResponseCode() < 300,
            statusCode: response.getResponseCode()
        };
    } catch (error) {
        logError('updateCounterRecord error: ' + error.toString());
        return { success: false, statusCode: 500 };
    }
}

/**
 * Check if visitor already has a record within 24 hours
 */
function checkExistingVisit(visitorId) {
    const today = getTodayDateISO();
    const encodedVisitorId = encodeURIComponent(visitorId);
    const url = `${SUPABASE_URL}/rest/v1/${TABLE_NAME}_visitor_log?visitor_id=eq.${encodedVisitorId}&visit_date=eq.${today}`;

    const options = {
        method: 'get',
        headers: {
            'apikey': SUPABASE_API_KEY,
            'Authorization': 'Bearer ' + SUPABASE_API_KEY,
            'Content-Type': 'application/json'
        },
        muteHttpExceptions: true
    };

    try {
        const response = UrlFetchApp.fetch(url, options);
        const data = JSON.parse(response.getContentText());
        return Array.isArray(data) && data.length > 0;
    } catch (error) {
        logError('checkExistingVisit error: ' + error.toString());
        // Fail open - don't record if check fails
        return false;
    }
}

// ===== UTILITY FUNCTIONS =====
/**
 * Log errors to Supabase for debugging
 */
function logError(message) {
    const url = `${SUPABASE_URL}/rest/v1/gas_error_logs`;
    const now = new Date().toISOString();

    const payload = {
        error_message: message,
        timestamp: now
    };

    const options = {
        method: 'post',
        headers: {
            'apikey': SUPABASE_API_KEY,
            'Authorization': 'Bearer ' + SUPABASE_API_KEY,
            'Content-Type': 'application/json'
        },
        payload: JSON.stringify(payload),
        muteHttpExceptions: true
    };

    try {
        UrlFetchApp.fetch(url, options);
    } catch (e) {
        // Silently fail if logging fails
    }

    // Also log to Apps Script console
    Logger.log('[ERROR] ' + message);
}

/**
 * Get ISO date string for 24 hours ago
 */
function get24HoursAgoISO() {
    const now = new Date();
    const hours24ago = new Date(now.getTime() - (24 * 60 * 60 * 1000));
    return hours24ago.toISOString();
}

/**
 * Get today's date (UTC) as YYYY-MM-DD
 */
function getTodayDateISO() {
    return new Date().toISOString().slice(0, 10);
}

/**
 * Create JSON response
 */
function createResponse(success, message, statusCode, data) {
    const response = ContentService.createTextOutput(JSON.stringify({
        success: success,
        message: message,
        data: data || null
    }));

    response.setMimeType(ContentService.MimeType.JSON);
    return response;
}
