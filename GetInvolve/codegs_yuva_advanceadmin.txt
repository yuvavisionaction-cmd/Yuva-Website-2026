// ===== MAIN HANDLER =====
/**
 * Handle incoming requests from the frontend
 */
function doPost(e) {
    try {
        let payload = {};
        
        // Try to parse form data first (for CORS bypass), then JSON
        if (e.parameter && Object.keys(e.parameter).length > 0) {
            // Form-urlencoded data
            payload = e.parameter;
            // Parse recipients if it's a comma-separated string
            if (payload.recipients && typeof payload.recipients === 'string') {
                payload.recipients = payload.recipients.split(',').map(r => r.trim());
            }
        } else {
            // JSON data
            payload = JSON.parse(e.postData.contents || '{}');
        }

        // Notification/email methods do not need counter visitor/session fields.
        if (payload.action === 'notify') {
            return handleNotify(payload);
        }

        // Validate counter request
        if (!validateRequest(payload)) {
            return createResponse(false, 'Invalid request', 400);
        }

        let result;

        switch (payload.action) {
            case 'recordRefresh':
                result = recordPageRefresh(payload);
                break;
            case 'recordUniqueVisit':
                result = recordUniqueVisitor(payload);
                break;
            case 'getCounters':
                result = getCounters(payload);
                break;
            case 'reset':
                result = resetCounters(payload);
                break;
            default:
                return createResponse(false, 'Unknown action: ' + payload.action, 400);
        }

        return createResponse(result.success, result.message, result.statusCode || 200, result.data);
    } catch (error) {
        logError('doPost error: ' + error.toString());
        return createResponse(false, 'Server error: ' + error.toString(), 500);
    }
}

// ===== NOTIFY / EMAIL HANDLER =====
function handleNotify(payload) {
    try {
        const method = String(payload.method || '').trim();

        switch (method) {
            case 'sendVimarshCertificateEmails':
                return sendVimarshCertificateEmails(payload);
            case 'sendBroadcastEmails':
                return sendBroadcastEmails(payload);
            case 'sendBatchEmails':
                return sendBatchEmails(payload);
            case 'sendIndividualEmail':
                return sendIndividualEmail(payload);
            default:
                return createResponse(false, 'Invalid notify method: ' + method, 400);
        }
    } catch (error) {
        logError('handleNotify error: ' + error.toString());
        return createResponse(false, 'Notify handler failed: ' + error.toString(), 500);
    }
}

}

/**

function doOptions(e) {
    const output = ContentService.createTextOutput('');
    output.setMimeType(ContentService.MimeType.TEXT);
    output.setHeaders({
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
        'Access-Control-Allow-Headers': 'Content-Type, Authorization',
        'Access-Control-Max-Age': '86400'

// ===== REQUEST VALIDATION =====
/**
 * Validate incoming request
 */
function validateRequest(payload) {
    // Check required fields
    if (!payload.action || !payload.visitorId || !payload.sessionToken) {
        return false;
    }

    // Check if visitor ID format is valid
    if (!payload.visitorId.toString().startsWith('visitor_')) {
        return false;
    }

    // Check if session token format is valid
    if (!payload.sessionToken.toString().startsWith('session_')) {
        return false;
    }

    return true;
}

// ===== DATABASE OPERATIONS =====
/**
 * Get the main counter record from Supabase
 */
function getCounterRecord() {
    const url = `${SUPABASE_URL}/rest/v1/${TABLE_NAME}?id=eq.1`;

    const options = {
        method: 'get',
        headers: {
            'apikey': SUPABASE_API_KEY,
            'Authorization': 'Bearer ' + SUPABASE_API_KEY,
            'Content-Type': 'application/json'
        },
        muteHttpExceptions: true
    };

    try {
        const response = UrlFetchApp.fetch(url, options);
        const data = JSON.parse(response.getContentText());

        if (Array.isArray(data) && data.length > 0) {
            return data[0];
        }

        // Create new record if doesn't exist
        return createCounterRecord();
    } catch (error) {
        logError('getCounterRecord error: ' + error.toString());
        throw error;
    }
}

/**
 * Get ISO date string for 24 hours ago
 */
function get24HoursAgoISO() {
    const now = new Date();
    const hours24ago = new Date(now.getTime() - (24 * 60 * 60 * 1000));
    return hours24ago.toISOString();
}

/**
 * Get today's date (UTC) as YYYY-MM-DD
 */
function getTodayDateISO() {
    return new Date().toISOString().slice(0, 10);
}

/**
 * Create JSON response - ContentService.TextOutput does NOT support setHeaders().
 * GAS deployed web apps handle CORS automatically; no manual header setting needed.
 */
function createResponse(success, message, statusCode, data) {
    const payload = {
        success: success,
        message: message || (success ? 'OK' : 'Error'),
        statusCode: statusCode || (success ? 200 : 500)
    };
    if (data !== undefined && data !== null) {
        payload.data = data;
    }
    return ContentService
        .createTextOutput(JSON.stringify(payload))
        .setMimeType(ContentService.MimeType.JSON);
}

            const batch = recipients.slice(i, i + BATCH_SIZE);
            
            for (let j = 0; j < batch.length; j++) {
                const email = String(batch[j]).trim();
                
                if (!email || email.length === 0) {
                    failCount++;
                    continue;
                }
                
                try {
                    GmailApp.sendEmail(email, subject, message, {
                        htmlBody: htmlBody,
                        name: 'YUVA Admin'
                    });

                    failCount++;
                    errors.push(email + ': ' + sendError.message);
                    logError('Failed to send email to ' + email + ': ' + sendError.toString());
                }
            }
            
            // Small delay between batches to avoid rate limiting
            Utilities.sleep(500);
        }
        
        // Update broadcast status in Supabase
        updateBroadcastStatus(broadcastId, successCount, failCount);
        
        return createResponse(true, 'Broadcast sent', 200, {
            sent: successCount,
            failed: failCount,
            total: recipients.length,
            broadcastId: broadcastId
        });
        
    } catch (error) {
        logError('sendBroadcastEmails error: ' + error.toString());
        return createResponse(false, 'Failed to send emails: ' + error.toString(), 500);
    }
}

function updateSingleEmailStatus(broadcastId, isSent) {
    try {
        const url = SUPABASE_URL + '/rest/v1/email_broadcasts?id=eq.' + broadcastId;
        const payload = {
            sent_count: isSent ? 1 : 0,
            failed_count: isSent ? 0 : 1,
            status: isSent ? 'sent' : 'failed',
            sent_at: new Date().toISOString()
        };

        const options = {
            method: 'PATCH',
            headers: {
                'apikey': SUPABASE_API_KEY,
                'Authorization': 'Bearer ' + SUPABASE_API_KEY,
                'Content-Type': 'application/json',
                'Prefer': 'return=representation'
            },
            payload: JSON.stringify(payload),
            muteHttpExceptions: true
        };

        const response = UrlFetchApp.fetch(url, options);
        const statusCode = response.getResponseCode();
        if (statusCode < 200 || statusCode >= 300) {
            logError('updateSingleEmailStatus failed: HTTP ' + statusCode + ' for broadcastId ' + broadcastId);
        }
    } catch (error) {
        logError('updateSingleEmailStatus error: ' + error.toString());
    }
}

/**
 * Update broadcast status in Supabase after sending
 * @param {number} broadcastId - The broadcast ID
 * @param {number} sentCount - Number of emails sent successfully
 * @param {number} failedCount - Number of emails that failed
 */
function updateBroadcastStatus(broadcastId, sentCount, failedCount) {
    try {
        const url = SUPABASE_URL + '/rest/v1/email_broadcasts?id=eq.' + broadcastId;
        
        const payload = {
            sent_count: sentCount,
            failed_count: failedCount,
            status: failedCount === 0 ? 'sent' : 'partial',
            sent_at: new Date().toISOString()
        };
        
        const options = {
            method: 'PATCH',
            headers: {
                'apikey': SUPABASE_API_KEY,
                'Authorization': 'Bearer ' + SUPABASE_API_KEY,
                'Content-Type': 'application/json',
                'Prefer': 'return=representation'
            },
            payload: JSON.stringify(payload),
            muteHttpExceptions: true
        };
        
        const response = UrlFetchApp.fetch(url, options);
        const statusCode = response.getResponseCode();
        
        if (statusCode < 200 || statusCode >= 300) {
            logError('updateBroadcastStatus failed: HTTP ' + statusCode);
        }
    } catch (error) {
        logError('updateBroadcastStatus error: ' + error.toString());
    }
}
