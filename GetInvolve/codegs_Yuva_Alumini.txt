/**
 * Handles POST requests from the alumni registration form
 */
function doPost(e) {
  try {
    // Check if this is an image upload request
    if (e.parameter.action === 'uploadImage') {
      return handleImageUpload(e);
    }
    
    // Parse form data
    const formData = parseFormData(e);
    
    // Validate required fields
    const validation = validateFormData(formData);
    if (!validation.valid) {
      return createResponse(false, validation.error);
    }
    
    // Check for duplicate email
    const duplicate = checkDuplicateEmail(formData.email);
    if (duplicate) {
      return createResponse(false, 'This email is already registered in our alumni network.');
    }
    
    // Prepare alumni data for Supabase
    const alumniData = {
      full_name: formData.fullName,
      email: formData.email.toLowerCase().trim(),
      phone: formData.phone,
      batch_year: formData.batch,
      current_city: formData.city,
      profession: formData.profession,
      organization: formData.organization,
      image_url: formData.imageUrl || null,
      interests: formData.interests ? formData.interests.split(',').map(i => i.trim()) : [],
      is_featured: false,
      is_approved: true, // Auto-approve all registrations
      badge: null,
      quote: null
    };
    
    // Insert into Supabase
    const result = insertAlumniToSupabase(alumniData);
    
    if (result.success) {
      // Send confirmation email (optional)
      if (SEND_CONFIRMATION_EMAIL) {
        sendConfirmationEmail(alumniData);
        notifyAdmin(alumniData);
      }
      
      return createResponse(true, 'Registration successful! Welcome to YUVA Alumni Network.', result.data);
    } else {
      throw new Error(result.error || 'Failed to save alumni data');
    }
    
  } catch (error) {
    Logger.log('Error in doPost: ' + error.toString());
    return createResponse(false, 'An error occurred during registration. Please try again.');
  }
}

/**
 * Handle image upload to Supabase Storage
 */
function handleImageUpload(e) {
  try {
    const imageData = e.parameter.imageData; // Base64 encoded image
    const fileName = e.parameter.fileName;
    const fileType = e.parameter.fileType;
    
    if (!imageData || !fileName) {
      return createResponse(false, 'Missing image data');
    }
    
    // Decode base64 to blob
    const imageBlob = Utilities.newBlob(
      Utilities.base64Decode(imageData),
      fileType,
      fileName
    );
    
    // Upload to Supabase Storage
    const filePath = `profiles/${fileName}`;
    const uploadUrl = `${SUPABASE_URL}/storage/v1/object/${STORAGE_BUCKET}/${filePath}`;
    
    const options = {
      method: 'post',
      headers: {
        'apikey': SUPABASE_SERVICE_KEY,
        'Authorization': `Bearer ${SUPABASE_SERVICE_KEY}`,
        'Content-Type': fileType
      },
      payload: imageBlob.getBytes(),
      muteHttpExceptions: true
    };
    
    const response = UrlFetchApp.fetch(uploadUrl, options);
    const responseCode = response.getResponseCode();
    
    if (responseCode === 200 || responseCode === 201) {
      // Generate public URL
      const publicURL = `${SUPABASE_URL}/storage/v1/object/public/${STORAGE_BUCKET}/${filePath}`;
      
      return createResponse(true, 'Image uploaded successfully', { imageUrl: publicURL });
    } else {
      const errorText = response.getContentText();
      Logger.log('Upload error: ' + errorText);
      return createResponse(false, 'Failed to upload image: ' + errorText);
    }
    
  } catch (error) {
    Logger.log('Image upload error: ' + error.toString());
    return createResponse(false, 'Image upload failed: ' + error.toString());
  }
}

/**
 * Handles GET requests from the frontend
 */
function doGet(e) {
  try {
    // Safety check for when running in Apps Script editor
    if (!e || !e.parameter) {
      return createResponse(false, 'Missing parameters. Use testGetFeaturedAlumni() to test in editor.');
    }
    
    const action = e.parameter.action;
    
    if (action === 'getFeaturedAlumni') {
      return getFeaturedAlumni();
    } else if (action === 'getAlumniByBatch') {
      const batch = e.parameter.batch;
      return getAlumniByBatch(batch);
    } else if (action === 'searchAlumni') {
      const query = e.parameter.query;
      return searchAlumni(query);
    } else {
      return createResponse(false, 'Invalid action');
    }
    
  } catch (error) {
    Logger.log('Error in doGet: ' + error.toString());
    return createResponse(false, 'An error occurred while fetching data.');
  }
}

/**
 * Handles OPTIONS requests for CORS pre-flight checks
 */
function doOptions(e) {
  return ContentService
    .createTextOutput(JSON.stringify({ status: 'SUCCESS' }))
    .setMimeType(ContentService.MimeType.JSON);
}

/**
 * Insert alumni data into Supabase
 */
function insertAlumniToSupabase(alumniData) {
  try {
    const url = `${SUPABASE_URL}/rest/v1/${SUPABASE_TABLE}`;
    
    const options = {
      method: 'post',
      headers: {
        'apikey': SUPABASE_SERVICE_KEY,
        'Authorization': `Bearer ${SUPABASE_SERVICE_KEY}`,
        'Content-Type': 'application/json',
        'Prefer': 'return=representation'
      },
      payload: JSON.stringify(alumniData),
      muteHttpExceptions: true
    };
    
    const response = UrlFetchApp.fetch(url, options);
    const responseCode = response.getResponseCode();
    const responseText = response.getContentText();
    
    if (responseCode === 201) {
      const data = JSON.parse(responseText);
      return { success: true, data: data[0] };
    } else {
      Logger.log('Supabase Error: ' + responseText);
      return { success: false, error: 'Database error: ' + responseText };
    }
    
  } catch (error) {
    Logger.log('Insert Error: ' + error.toString());
    return { success: false, error: error.toString() };
  }
}

/**
 * Check if email already exists in Supabase
 */
function checkDuplicateEmail(email) {
  try {
    const url = `${SUPABASE_URL}/rest/v1/${SUPABASE_TABLE}?email=eq.${encodeURIComponent(email.toLowerCase().trim())}&select=id`;
    
    const options = {
      method: 'get',
      headers: {
        'apikey': SUPABASE_SERVICE_KEY,
        'Authorization': `Bearer ${SUPABASE_SERVICE_KEY}`
      },
      muteHttpExceptions: true
    };
    
    const response = UrlFetchApp.fetch(url, options);
    const data = JSON.parse(response.getContentText());
    
    return data.length > 0;
    
  } catch (error) {
    Logger.log('Duplicate check error: ' + error.toString());
    return false; // Allow registration if check fails
  }
}

/**
 * Search alumni by name, city, or profession
 */
function searchAlumni(query) {
  try {
    // Supabase text search using ilike (case-insensitive)
    const searchQuery = encodeURIComponent(`%${query}%`);
    const url = `${SUPABASE_URL}/rest/v1/${SUPABASE_TABLE}?is_approved=eq.true&or=(full_name.ilike.${searchQuery},current_city.ilike.${searchQuery},profession.ilike.${searchQuery})&select=*&order=full_name.asc&limit=20`;
    
    const options = {
      method: 'get',
      headers: {
        'apikey': SUPABASE_SERVICE_KEY,
        'Authorization': `Bearer ${SUPABASE_SERVICE_KEY}`
      },
      muteHttpExceptions: true
    };
    
    const response = UrlFetchApp.fetch(url, options);
    const data = JSON.parse(response.getContentText());
    
    return createResponse(true, 'Search results', data);
    
  } catch (error) {
    Logger.log('Search alumni error: ' + error.toString());
    return createResponse(false, 'Search failed');
  }
}

/**
 * Parse form data from POST request
 */
function parseFormData(e) {
  const params = e.parameter;
  return {
    fullName: params.fullName || '',
    email: params.email || '',
    phone: params.phone || '',
    batch: params.batch || '',
    city: params.city || '',
    profession: params.profession || '',
    organization: params.organization || '',
    imageUrl: params.imageUrl || '',
    interests: params.interests || ''
  };
}

/**
 * Validate form data
 */
function validateFormData(data) {
  const required = ['fullName', 'email', 'phone', 'batch', 'city', 'profession', 'organization'];
  
  for (let field of required) {
    if (!data[field] || data[field].trim() === '') {
      return { valid: false, error: `${field} is required` };
    }
  }
  
  // Validate email format
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  if (!emailRegex.test(data.email)) {
    return { valid: false, error: 'Invalid email format' };
  }
  
  // Validate phone (basic check)
  if (data.phone.length < 10) {
    return { valid: false, error: 'Invalid phone number' };
  }
  
  return { valid: true };
}

/**

function createResponse(success, message, data = null) {
